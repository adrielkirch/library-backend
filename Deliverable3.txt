1. Get all books by a specific author


    SELECT * FROM Books WHERE author_id = 'specific_author_id';

2. Get the total number of books sold

    SELECT SUM(quantity) AS total_books_sold FROM OrderDetails;

3. Get the total revenue generated from orders

   SELECT SUM(quantity * price) AS total_revenue FROM OrderDetails JOIN Books ON OrderDetails.book_id = Books.book_id;

4. Get the top 5 best-selling books

    SELECT Books.title, SUM(OrderDetails.quantity) AS total_sold
    FROM OrderDetails
    JOIN Books ON OrderDetails.book_id = Books.book_id
    GROUP BY Books.book_id
    ORDER BY total_sold DESC
    LIMIT 5;


5. Get the customers who have spent the most

    SELECT Customers.name, SUM(OrderDetails.quantity * Books.price) AS total_spent
    FROM OrderDetails
    JOIN Orders ON OrderDetails.order_id = Orders.order_id
    JOIN Customers ON Orders.customer_id = Customers.customer_id
    JOIN Books ON OrderDetails.book_id = Books.book_id
    GROUP BY Customers.customer_id
    ORDER BY total_spent DESC;
